```javascript
class EventDrivenWebSocket {
    constructor() {
        this.ws = null;
        this.connectionId = null;
        this.reconnectAttempts = 0;
        this.maxReconnectAttempts = 5;
        this.heartbeatInterval = null;
    }

    connect() {
        this.ws = new WebSocket('ws://localhost:8765/ws/v2');
        this.setupEventHandlers();
        this.startHeartbeat();
    }

    setupEventHandlers() {
        this.ws.onmessage = (event) => {
            const message = JSON.parse(event.data);
            this.handleMessage(message);
        };
    }

    handleMessage(message) {
        switch(message.type) {
            case 'session_updated':
                this.handleSessionUpdate(message);
                break;
            case 'connection_established':
                this.handleConnectionEstablished(message);
                break;
            case 'state_changed':
                this.handleStateChange(message);
                break;
        }
    }

    handleSessionUpdate(message) {
        // 智能局部更新 - 解決核心問題
        this.updateSessionInfo(message.session_info);
        this.updateStatusDisplay(message);
        this.showUpdateNotification();
    }
}
```

```python
class OptimizedConnectionPool:
    def __init__(self, config):
        self.max_connections = config.max_connections
        self.connection_pool = asyncio.Queue(maxsize=self.max_connections)
        self.batch_size = 10
        self.batch_timeout = 0.1

    async def batch_broadcast(self, session_id: str, messages: List[Dict]):
        """批量廣播消息，提升性能"""
        connections = await self.get_session_connections(session_id)

        # 批量處理消息
        for i in range(0, len(messages), self.batch_size):
            batch = messages[i:i + self.batch_size]
            await self._send_batch(connections, batch)

    async def _send_batch(self, connections, messages):
        """批量發送消息到多個連接"""
        tasks = []
        for connection in connections:
            for message in messages:
                tasks.append(connection.websocket.send_json(message))

        await asyncio.gather(*tasks, return_exceptions=True)
```

```python
class PerformanceMonitor:
    def __init__(self):
        self.metrics = {
            'events_processed': 0,
            'connections_active': 0,
            'memory_usage': 0,
            'cpu_usage': 0,
            'response_times': deque(maxlen=1000)
        }

    async def collect_metrics(self):
        """收集系統性能指標"""
        process = psutil.Process()

        self.metrics.update({
            'memory_usage': process.memory_info().rss / 1024 / 1024,  # MB
            'cpu_usage': process.cpu_percent(),
            'connections_active': len(connection_pool.connections),
            'timestamp': time.time()
        })

    async def health_check_enhanced(self):
        """增強的健康檢查"""
        health_status = {
            'status': 'healthy',
            'components': {},
            'metrics': self.metrics,
            'alerts': []
        }

        # 檢查各組件健康狀態
        components = [
            ('event_bus', event_bus),
            ('state_manager', state_manager),
            ('connection_pool', connection_pool)
        ]

        for name, component in components:
            try:
                status = await component.health_check()
                health_status['components'][name] = status
            except Exception as e:
                health_status['components'][name] = {'status': 'error', 'error': str(e)}
                health_status['alerts'].append(f'{name} 組件異常: {e}')

        return health_status
```
